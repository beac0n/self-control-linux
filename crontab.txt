# linux-focus-tools crontab entries
#
# Add the user entries with: crontab -e
# Add the root entry with:   sudo crontab -e
#
# Replace 1000 with your actual UID (check with: id -u)
# Replace the times to match your preferred shutdown time.
# The example below shuts down at 23:00 with warnings at 22:50, 22:55, 22:59.

# --- USER crontab (crontab -e) ---
50 22 * * * DISPLAY=:0 DBUS_SESSION_BUS_ADDRESS=unix:path=/run/user/1000/bus notify-send -u critical "⚠️ Shutdown in 10 minutes" "System will shut down at 23:00"
55 22 * * * DISPLAY=:0 DBUS_SESSION_BUS_ADDRESS=unix:path=/run/user/1000/bus notify-send -u critical "⚠️ Shutdown in 5 minutes" "System will shut down at 23:00"
59 22 * * * DISPLAY=:0 DBUS_SESSION_BUS_ADDRESS=unix:path=/run/user/1000/bus notify-send -u critical "⚠️ Shutdown in 1 minute" "System will shut down at 23:00"

# --- ROOT crontab (sudo crontab -e) ---
0 23 * * * /sbin/shutdown -h now
